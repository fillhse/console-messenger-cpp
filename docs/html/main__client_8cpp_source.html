50/**
51 * @brief Проверить корректность IPv4-адреса и порта.
52 *
53 * Использует inet_pton() для валидации формата IPv4
54 * и проверяет, что порт находится в диапазоне 1..65535.
55 *
56 * @param ip Строка с IPv4-адресом.
57 * @param port Номер порта.
58 * @return true если адрес и порт валидны;
59 * false в противном случае.
60 */
61bool valid_ip_port(const std::string& ip, int port) {
62 sockaddr_in tmp{};
63 return inet_pton(AF_INET, ip.c_str(), &tmp.sin_addr) == 1 && port > 0 && port < 65536;
64}

66/**
67 * @brief Считать или запросить у пользователя настройки сервера.
68 *
69 * Если файл с конфигурацией существует, пытается прочитать из него строку
70 * в формате "IP:порт". Если данные некорректны или файла нет,
71 * запрашивает ввод у пользователя до тех пор, пока не будет введена

73 * Сохраняет корректные настройки в файл.
74 *
75 * @return Настройки сервера в виде ServerConf.

77ServerConf get_config() {
78 std::filesystem::create_directories(CFG_DIR);
79 std::ifstream fin(CFG_FILE);
80 std::string ip;
81 int port;
82 bool ok = false;
83 if (fin) {
84 std::string line;
85 std::getline(fin, line);
86 std::istringstream ss(line);
87 std::getline(ss, ip, ':');
88 ss >> port;
89 ok = valid_ip_port(ip, port);
90 }
91 while (!ok) {
92 std::cout << "Enter server IP: ";
93 std::cin >> ip;
94 std::cout << "Enter server port: ";
95 std::cin >> port;
96 std::cin.ignore();
97 ok = valid_ip_port(ip, port);
98 if (!ok)
99 std::cout << "Invalid IP or port. Try again.\n";
100 }
101 std::ofstream(CFG_FILE, std::ios::trunc) << ip << ':' << port << '\n';
102 return {ip, port};
103}

105/**
106 * @brief Цикл приёма и вывода сообщений от сервера.
107 *
108 * Читает строки из сокета через recv_line() до разрыва соединения.
109 * Выводит каждую строку на консоль. При получении специального
110 * маркера "*ENDM*" отображает приглашение ввода.
111 *
112 * @param fd Дескриптор подключённого сокета сервера.
113 */
114void receive_messages(int fd) {
115 std::string line;
116 while (recv_line(fd, line)) {
117 if (line.empty())
118 continue;
119 if (line == "*ENDM*") {
120 std::cout << "[You]> " << std::flush;
121 continue;
122 }
123 std::cout << line << '\n';
124 }
125 std::cout << "\nDisconnected from server.\n";
126 close(fd);
127 exit(0);
128}

130/**
131 * @brief Точка входа клиентского приложения.
132 *
133 * Получает конфигурацию сервера, устанавливает TCP-соединение,
134 * запускает поток для приёма сообщений и в цикле
135 * отправляет введённые пользователем сообщения.
136 *
137 * @return Код завершения (0 при успехе, иначе 1).
138 */
139int main() {
140 ServerConf conf = get_config();
141
142 int sock = socket(AF_INET, SOCK_STREAM, 0);
143 if (sock == -1) {
144 perror("socket");
145 return 1;
146 }
147
148 sockaddr_in addr{};
149 addr.sin_family = AF_INET;
150 addr.sin_port = htons(conf.port);
151 inet_pton(AF_INET, conf.ip.c_str(), &addr.sin_addr);
152
153 if (connect(sock, (sockaddr*)&addr, sizeof(addr)) < 0) {
154 perror("connect");
155 return 1;
156 }
157
158 std::thread(receive_messages, sock).detach();
159
160 std::string input;
161 while (std::getline(std::cin, input)) {
162 if (input.empty())
163 continue;
164 if (input.size() > MAX_INPUT) {
165 std::cout << "Message longer than 2000 characters. Split it.\n";
166 continue;
167 }
168 if (input == "/exit") {
169 send_line(sock, "/exit");
170 std::cout << "\nExiting...\n";
171 break;
172 }
173 send_line(sock, input);
174 }
175 close(sock);
176 return 0;
177}
