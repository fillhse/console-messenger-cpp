114bool send_telegram_code(const std::string& chat_id, const std::string& code) {
115 ensure_bot_token();
116 std::string url = "https://api.telegram.org/bot" + BOT_TOKEN + "/sendMessage";
117 std::string cmd = "curl -s -X POST \"" + url +
118 "\""
119 " -d chat_id=" +
120 chat_id + " -d text='Your authentication code is: " + code + "'";
121
122 FILE* pipe = popen(cmd.c_str(), "r");
123 if (!pipe)
124 return false;
125
126 std::string response;
127 char buf[256];
128 while (fgets(buf, sizeof(buf), pipe))
129 response += buf;
130 int status = pclose(pipe);
131
132 if (status == 0 && response.find("\"ok\":true") != std::string::npos) {
133 auth_codes[chat_id] = code;
134 return true;
135 }
136 return false;
137}
